## Diff Semântico entre BIRs

```rust
fn diff_bir(original: &BirDevice, modern: &BirDevice) -> EvolutionReport {
    let mut preserved = 0u32;
    let mut broken = Vec::new();

    // Verifica cada contrato original no novo hardware
    for contract in &original.contracts {
        if let Some(modern_contract) = modern.find_similar_contract(&contract) {
            if contract.compatible_with(&modern_contract) {
                preserved += 1;
            } else {
                broken.push(Incompatibility {
                    contract: contract.clone(),
                    reason: format!(
                        "latency {}..{} → {}..{} (violates max {})",
                        contract.latency.min, contract.latency.max,
                        modern_contract.latency.min, modern_contract.latency.max,
                        contract.latency.max,
                    ),
                });
            }
        }
    }

    EvolutionReport {
        title: format!("{} → {}", original.name, modern.name),
        contracts_preserved: preserved,
        contracts_broken: broken.len(),
        details: broken,
    }
}
```
